target_epus <- c("MAB", "GB", "GOM")

purrr::walk(target_epus, function(epu_name) {
  strata_codes <- epu_strata[[epu_name]]
  
  dat <- survdat::calc_stratified_mean(
    data |> dplyr::mutate(group = "unit"), 
    filterByArea = strata_codes,
    filterBySeason = c('FALL', 'SPRING'),
    groupDescription = 'group',
    tidy = TRUE
  ) 

  plt <- dat |>
    dplyr::filter(variable == "strat.biomass") |>
    ggplot2::ggplot(ggplot2::aes(x = YEAR, y = value, color = SEASON)) +
    ggplot2::geom_line() +
    ggplot2::geom_point() +
    ggplot2::theme_bw() +
    ggplot2::ggtitle(paste(epu_name, "stratified mean biomass")) 
  print(plt)
})

purrr::walk(target_epus, function(epu_name) {
  strata_codes <- epu_strata[[epu_name]]
  
  # Build a combined grouping factor for calculation
  guild_prep <- joined_dat |> 
    dplyr::mutate(
      Status = ifelse(Fed.Managed == "Unmanaged", "Unmanaged", "Managed"),
      guild_cat = paste0(SOE.24, "___", Status)
    ) |> 
    dplyr::filter(!is.na(SOE.24) & SOE.24 != "NA")
  
  dat <- survdat::calc_stratified_mean(
    guild_prep,
    filterByArea = strata_codes,
    filterBySeason = c('FALL', 'SPRING'),
    groupDescription = 'guild_cat',
    tidy = TRUE
  )
  
  # Separate back out into explicit columns for plotting
  processed_dat <- dat |> 
    dplyr::filter(variable == "strat.biomass") |> 
    tidyr::separate(guild_cat, into = c("Guild", "Management_Status"), sep = "___")
  
  # Walk through each unique guild to generate a separate plot
  unique_guilds <- unique(processed_dat$Guild)
  
  purrr::walk(unique_guilds, function(guild_name) {
    dat <- processed_dat |> 
      dplyr::filter(Guild == guild_name) 
    
    if(nrow(dat ) > 4) {
    plt <- dat |> 
      ggplot2::ggplot(ggplot2::aes(x = YEAR, y = value, color = Management_Status)) + 
      ggplot2::geom_line() + 
      ggplot2::geom_point() + 
      ggplot2::theme_bw() + 
      ggplot2::facet_wrap(~SEASON, scales = "free_y") + 
      ggplot2::labs(
        title = paste(epu_name, "-", guild_name, "Stratified Mean Biomass"),
        y = "Stratified Mean Biomass",
        color = "Status"
      )
    
    print(plt)
    }
  })
})

Guild Biomass broken out by managed species and Unmanaged

Here we evaluate each guild in its own standalone plot, displaying the breakdown of individual species biomass contributions alongside a unified block representing all Unmanaged species combined. Species contributing less than 1% of the total biomass in a given year/season are grouped into an “Other” category to clean up the legend.

Code 
purrr::walk(target_epus, function(epu_name) {
  strata_codes <- epu_strata[[epu_name]]
  
  # Bake the metadata into the grouping string for the data.table aggregation
  strat_prep <- joined_dat |>
    dplyr::mutate(
      cat = paste(SOE.24, Fed.Managed, COMNAME, sep = "___")
    )
  
  dat <- survdat::calc_stratified_mean(
    strat_prep,
    filterByArea = strata_codes,
    filterBySeason = c('FALL', 'SPRING'),
    groupDescription = 'cat',
    tidy = TRUE
  ) 
  
  # Post-process the aggregated dataset
  processed_dat <- dat |>
    dplyr::filter(variable == "strat.biomass") |>
    tidyr::separate(cat, into = c("Guild", "Fed.Managed", "COMNAME"), sep = "___") |>
    dplyr::filter(!is.na(Guild) & Guild != "NA") |>
    # Initial classification for Unmanaged vs Managed names
    dplyr::mutate(
      Plot_Group = ifelse(Fed.Managed == "Unmanaged", "Unmanaged", COMNAME)
    ) |>
    # Calculate annual percentage contribution per Guild and Season
    dplyr::group_by(YEAR, SEASON, Guild) |>
    dplyr::mutate(
      Total_Guild_Biomass = sum(value, na.rm = TRUE),
      Pct_Contribution = ifelse(Total_Guild_Biomass > 0, value / Total_Guild_Biomass, 0),
      # If contribution is 1% or less, reassign group name to "Other"
      Plot_Group = ifelse(Pct_Contribution <= 0.01, "Other", Plot_Group)
    ) |>
    dplyr::ungroup()
  
  unique_guilds <- unique(processed_dat$Guild)
  
  # Walk through each guild to yield a standalone plot
  purrr::walk(unique_guilds, function(guild_name) {
    
    # Filter for the current guild and ensure "Other" is factored at the end if it exists
    guild_data <- processed_dat |> 
      dplyr::filter(Guild == guild_name)
    
    # Optional: Clean up the legend factor levels so "Other" and "Unmanaged" sit at the bottom
    all_groups <- unique(guild_data$Plot_Group)
    clean_levels <- c(setdiff(all_groups, c("Unmanaged", "Other")), 
                      intersect(all_groups, "Unmanaged"), 
                      intersect(all_groups, "Other"))
    
    guild_data <- guild_data |> 
      dplyr::mutate(Plot_Group = factor(Plot_Group, levels = clean_levels))
    
    if(nrow(guild_data) > 4) {
         plt <- guild_data |>
      ggplot2::ggplot(ggplot2::aes(x = YEAR, y = value, fill = Plot_Group)) +
      ggplot2::geom_col() +
      viridis::scale_fill_viridis(discrete = TRUE) +
      ggplot2::theme_bw() +
      ggplot2::facet_wrap(~SEASON, scales = "free_y") +
      ggplot2::labs(
        title = paste0(epu_name, " - ", guild_name, " Biomass Breakout"),
        subtitle = "Species contributing <= 1% annually grouped into 'Other'",
        y = "Stratified Mean Biomass",
        fill = "Species/Group"
      ) +
      ggplot2::theme(legend.position = "bottom")
    
    print(plt)
    }
 
  })
})
